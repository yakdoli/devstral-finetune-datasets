// Custom comparer function.
public int Compare(object x, object y)
{
    if (x == null)
        return -1;
    else if (y == null)
        return 100;
    else
    {
        int i = int.Parse(x.ToString());
        int j = int.Parse(y.ToString());
        return i - j;
    }
}

// Custom categorizer function.
public static int GetCategory(int i)
{
    int ret = 0;
    if (i < 10)
        ret = 1;
    else if (i >= 10 && i < 20)
        ret = 2;
    else if (i >= 20 && i < 30)
        ret = 3;
    else if (i >= 30 && i < 40)
        ret = 4;
    else
        ret = 5;
    return ret;
}
