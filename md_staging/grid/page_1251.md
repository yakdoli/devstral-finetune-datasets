Private Sub SetupCheckBox()

            GridCellSizeKind.ActualSize)

                                  cellRect.Y + cellRect.Height \ 2 - 13 \ 2) ' 
    cellRect.Location;

End Sub

    SetupCheckBox()

End Sub
